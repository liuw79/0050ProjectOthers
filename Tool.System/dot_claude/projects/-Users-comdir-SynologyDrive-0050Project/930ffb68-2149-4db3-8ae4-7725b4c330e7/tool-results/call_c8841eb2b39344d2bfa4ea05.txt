   156→    const API_BASE = 'http://localhost:3000/api';
   157→    let currentUserId = null;
   158→
   159→    // 从localStorage获取userId
   160→    function getUserId() {
   161→      const userStr = localStorage.getItem('currentUser');
   162→      if (userStr) {
   163→        const user = JSON.parse(userStr);
   164→        return user.userId;
   165→      }
   166→      // 未登录则从URL获取（用于测试）
   167→      const params = new URLSearchParams(window.location.search);
   168→      return params.get('userId');
   169→    }
   170→
   171→    // 加载素材列表
   172→    async function loadMaterials() {
   173→      currentUserId = getUserId();
   174→      const tagFilter = document.getElementById('tagFilter').value;
   175→      const categoryFilter = document.getElementById('categoryFilter').value;
   176→
   177→      const container = document.getElementById('materialsList');
   178→      container.innerHTML = '<div class="loading">加载中...</div>';
   179→
   180→      try {
   181→        const params = new URLSearchParams({ userId: currentUserId });
   182→        if (tagFilter) params.append('tag', tagFilter);
   183→        if (categoryFilter) params.append('category', categoryFilter);
   184→
   185→        const response = await fetch(`${API_BASE}/materials?${params}`);
   186→        const data = await response.json();
   187→
   188→        if (data.success && data.materials.length > 0) {
   189→          container.innerHTML = data.materials.map(m => `
   190→            <div class="material-card">
   191→              <h3>${escapeHtml(m.title)}</h3>
   192→              <div class="content">${escapeHtml(m.content)}</div>
   193→              <div class="meta">
   194→                ${(m.tags || []).map(t => `<span class="tag">#${escapeHtml(t)}</span>`).join('')}
   195→                <span class="category">${escapeHtml(m.category)}</span>
   196→                <span class="date">${new Date(m.created_at).toLocaleDateString()}</span>
   197→              </div>
   198→            </div>
   199→          `).join('');
   200→        } else {
   201→          container.innerHTML = '<div class="empty">暂无素材，点击上方"添加素材"按钮创建</div>';
   202→        }
   203→      } catch (error) {
   204→        container.innerHTML = `<div class="empty">加载失败：${error.message}</div>`;
   205→      }
   206→    }
   207→
   208→    // HTML转义，防止XSS
   209→    function escapeHtml(text) {
   210→      const div = document.createElement('div');
   211→      div.textContent = text;
   212→      return div.innerHTML;
   213→    }
