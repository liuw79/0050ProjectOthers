     1→// 飞书文档API客户端
     2→const path = require('path');

     4→
     5→const FEISHU_BASE_URL = 'https://open.feishu.cn/open-apis';
     6→
     7→class FeishuDocClient {
     8→  constructor() {
     9→    this.appId = process.env.LARK_APP_ID;
    10→    this.appSecret = process.env.LARK_APP_SECRET;
    11→    this.appAccessToken = null;
    12→    this.tokenExpireTime = 0;
    13→  }
    14→
    15→  // 获取app access token
    16→  async getAppAccessToken() {
    17→    if (this.appAccessToken && Date.now() < this.tokenExpireTime) {
    18→      return this.appAccessToken;
    19→    }
    20→
    21→    try {
    22→      const response = await fetch(`${FEISHU_BASE_URL}/auth/v3/app_access_token/internal`, {
    23→        method: 'POST',
    24→        headers: { 'Content-Type': 'application/json' },
    25→        body: JSON.stringify({
    26→          app_id: this.appId,
    27→          app_secret: this.appSecret,
    28→        }),
    29→      });
    30→
    31→      const data = await response.json();
    32→      if (data.code !== 0) {
    33→        throw new Error(`获取app token失败: ${data.msg}`);
    34→      }
    35→
    36→      this.appAccessToken = data.app_access_token;
    37→      this.tokenExpireTime = Date.now() + (data.expire_in - 300) * 1000;
    38→      return this.appAccessToken;
    39→    } catch (error) {
    40→      console.error('获取飞书token失败:', error);
    41→      throw error;
    42→    }
    43→  }
    44→
    45→  // 创建文档
    46→  async createDocument(title, content, folderToken = '') {
    47→    try {
    48→      const token = await this.getAppAccessToken();
    49→
    50→      const response = await fetch(`${FEISHU_BASE_URL}/docx/v1/documents`, {
    51→        method: 'POST',
    52→        headers: {
    53→          'Authorization': `Bearer ${token}`,
    54→          'Content-Type': 'application/json',
    55→        },
    56→        body: JSON.stringify({
    57→          title: title,
    58→          folder_token: folderToken,
    59→          content: {
    60→            text: {
    61→              content: content,
    62→            }
    63→          }
    64→        }),
    65→      });
    66→
    67→      const data = await response.json();
    68→      if (data.code !== 0) {
    69→        throw new Error(`创建文档失败: ${data.msg}`);
    70→      }
    71→
    72→      return {
    73→        documentId: data.document.document_id,
    74→        revisionId: data.document.revision_id,
    75→      };
    76→    } catch (error) {
    77→      console.error('创建文档失败:', error);
    78→      throw error;
    79→    }
    80→  }
    81→

    87→      const response = await fetch(`${FEISHU_BASE_URL}/docx/v1/documents/${documentId}/raw_content`, {
    88→        method: 'GET',
    89→        headers: {
    90→          'Authorization': `Bearer ${token}`,
    91→        },
    92→      });
    93→
    94→      const text = await response.text();

    96→    } catch (error) {
    97→      console.error('读取文档失败:', error);

    99→    }
   100→  }

   130→  // 设置文档仅管理员可见
   131→  async setAdminOnly(documentId, adminUserIds) {
