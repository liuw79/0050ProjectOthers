   260→  <script>
   261→    const API_BASE = 'http://localhost:3000/api';
   262→    let currentUserId = null;
   263→    let tags = [];
   264→    let fileContent = '';
   265→
   266→    // 从localStorage获取userId
   267→    function getUserId() {
   268→      const userStr = localStorage.getItem('currentUser');
   269→      if (userStr) {
   270→        const user = JSON.parse(userStr);
   271→        return user.userId || user.openId;
   272→      }
   273→      const params = new URLSearchParams(window.location.search);
   274→      return params.get('userId');
   275→    }
   276→
   277→    // 添加标签
   278→    function addTag() {
   279→      const input = document.getElementById('tagInput');
   280→      const value = input.value.trim();
   281→      if (value && !tags.includes(value)) {
   282→        tags.push(value);
   283→        renderTags();
   284→        input.value = '';
   285→      }
   286→    }
   287→
   288→    // 删除标签
   289→    function removeTag(index) {
   290→      tags.splice(index, 1);
   291→      renderTags();
   292→    }
   293→
   294→    // 渲染标签
   295→    function renderTags() {
   296→      const container = document.getElementById('tags');
   297→      container.innerHTML = tags.map((tag, index) => `
   298→        <span class="tag" onclick="removeTag(${index})">
   299→          ${escapeHtml(tag)}
   300→          <span class="remove">×</span>
   301→        </span>
   302→      `).join('');
   303→    }
   304→
   305→    // 显示消息
   306→    function showMessage(text, type) {
   307→      const message = document.getElementById('message');
   308→      message.textContent = text;
   309→      message.className = `message ${type}`;
   310→      message.style.display = 'block';
   311→      setTimeout(() => {
   312→        message.style.display = 'none';
   313→      }, 3000);
   314→    }
   315→
   316→    // 文件选择
   317→    function handleFileSelect(event) {
   318→      const file = event.target.files[0];
   319→      if (!file) return;
   320→
   321→      const reader = new FileReader();
   322→      reader.onload = (e) => {
   323→        fileContent = e.target.result;
   324→        document.getElementById('content').value = fileContent;
   325→        document.getElementById('filePreview').textContent = `已选择文件: ${file.name} (${Math.round(file.size / 1024)}KB)`;
   326→        document.getElementById('filePreview').style.display = 'block';
   327→      };
   328→      reader.readAsText(file);
   329→    }
   330→
   331→    // 拖放处理
   332→    const dropZone = document.getElementById('dropZone');
   333→    dropZone.addEventListener('dragover', (e) => {
   334→      e.preventDefault();
   335→      dropZone.classList.add('dragover');
   336→    });
   337→    dropZone.addEventListener('dragleave', () => {
   338→      dropZone.classList.remove('dragover');
   339→    });
   340→    dropZone.addEventListener('drop', (e) => {
   341→      e.preventDefault();
   342→      dropZone.classList.remove('dragover');
   343→      const file = e.dataTransfer.files[0];
   344→      if (file) {
   345→        document.getElementById('fileInput').files = e.dataTransfer.files;
   346→        handleFileSelect({ target: { files: [file] } });
   347→      }
   348→    });
   349→
   350→    // 导入课程
   351→    async function importCourse() {
   352→      currentUserId = getUserId();
   353→
   354→      const courseName = document.getElementById('courseName').value.trim();
   355→      const instructor = document.getElementById('instructor').value.trim();
   356→      const category = document.getElementById('category').value;
   357→      const content = document.getElementById('content').value.trim();
   358→
   359→      if (!courseName || !content) {
   360→        showMessage('请填写课程名称和内容', 'error');
   361→        return;
   362→      }
   363→
   364→      const form = document.querySelector('.form');
   365→      form.style.opacity = '0.5';
   366→      form.style.pointerEvents = 'none';
   367→
   368→      try {
   369→        const response = await fetch(`${API_BASE}/materials`, {
   370→          method: 'POST',
   371→          headers: { 'Content-Type': 'application/json' },
   372→          body: JSON.stringify({
   373→            userId: currentUserId,
   374→            title: courseName,
   375→            content: content + (instructor ? `\n\n讲师：${instructor}` : ''),
   376→            tags: tags,
   377→            category: category
   378→          })
   379→        });
   380→
   381→        const data = await response.json();
   382→
   383→        if (data.success) {
   384→          showMessage('课程导入成功！', 'success');
   385→          // 清空表单
   386→          document.getElementById('courseName').value = '';
   387→          document.getElementById('instructor').value = '';
   388→          document.getElementById('content').value = '';
   389→          document.getElementById('filePreview').style.display = 'none';
   390→          fileContent = '';
   391→          tags = [];
   392→          renderTags();
   393→        } else {
   394→          showMessage(data.error || '导入失败', 'error');
   395→        }
   396→      } catch (error) {
   397→        showMessage(`导入失败：${error.message}`, 'error');
   398→      } finally {
   399→        form.style.opacity = '1';
   400→        form.style.pointerEvents = 'auto';
   401→      }
   402→    }
   403→
   404→    // 表单提交（阻止默认行为）
   405→    function handleSubmit(event) {
   406→      event.preventDefault();
   407→    }
   408→
   409→    // 返回
   410→    function goBack() {
   411→      window.location.href = `materials.html?userId=${getUserId()}`;
   412→    }
   413→
   414→    // HTML转义
   415→    function escapeHtml(text) {
   416→      const div = document.createElement('div');
   417→      div.textContent = text;
   418→      return div.innerHTML;
   419→    }
   420→
   421→    // 标签输入回车处理
   422→    document.getElementById('tagInput').addEventListener('keydown', function(event) {
   423→      if (event.key === 'Enter') {
   424→        event.preventDefault();
   425→        addTag();
   426→      }
   427→    });
   428→
   429→    // 初始化
   430→    currentUserId = getUserId();
   431→  </script>
