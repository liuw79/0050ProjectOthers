     1→// 飞书OAuth客户端
     2→const path = require('path');
     3→require('dotenv').config({ path: path.join(__dirname, '../.env') });
     4→
     5→const FEISHU_BASE_URL = 'https://open.feishu.cn/open-apis';
     6→
     7→class FeishuAuthClient {
     8→  constructor() {
     9→    this.appId = process.env.LARK_APP_ID;
    10→    this.appSecret = process.env.LARK_APP_SECRET;
    11→    this.redirectUri = process.env.LARK_REDIRECT_URI || 'http://localhost:3000/auth/callback';
    12→  }
    13→
    14→  // 获取app access token
    15→  async getAppAccessToken() {
    16→    const response = await fetch(`${FEISHU_BASE_URL}/auth/v3/app_access_token/internal`, {
    17→      method: 'POST',
    18→      headers: { 'Content-Type': 'application/json' },
    19→      body: JSON.stringify({
    20→        app_id: this.appId,
    21→        app_secret: this.appSecret,
    22→      }),
    23→    });
    24→
    25→    const data = await response.json();
    26→    if (data.code !== 0) {
    27→      throw new Error(`获取app token失败: ${data.msg}`);
    28→    }
    29→
    30→    return data.app_access_token;
    31→  }
    32→
    33→  // 获取用户access token
    34→  async getUserAccessToken(authorizationCode) {
    35→    // 先获取 app_access_token
    36→    const appAccessToken = await this.getAppAccessToken();
    37→
    38→    // 再用 app_access_token 换取 user_access_token
    39→    const response = await fetch(`${FEISHU_BASE_URL}/authen/v1/access_token`, {
    40→      method: 'POST',
    41→      headers: {
    42→        'Content-Type': 'application/json',
    43→        'Authorization': `Bearer ${appAccessToken}`,
    44→      },
    45→      body: JSON.stringify({
    46→        grant_type: 'authorization_code',
    47→        redirect_uri: this.redirectUri,
    48→        code: authorizationCode,
    49→      }),
    50→    });
    51→
    52→    const data = await response.json();
    53→    if (data.code !== 0) {
    54→      throw new Error(`获取用户token失败: ${data.msg}`);
    55→    }
    56→
    57→    return {
    58→      accessToken: data.data.access_token,
    59→      refreshToken: data.data.refresh_token,
    60→      expiresIn: data.data.expires_in,
    61→    };
    62→  }
    63→
    64→  // 获取用户信息
    65→  async getUserInfo(accessToken) {
    66→    const response = await fetch(`${FEISHU_BASE_URL}/authen/v1/user_info`, {
    67→      method: 'GET',
    68→      headers: {
    69→        'Authorization': `Bearer ${accessToken}`,
    70→        'Content-Type': 'application/json',
    71→      },
    72→    });
    73→
    74→    const data = await response.json();
    75→    if (data.code !== 0) {
    76→      throw new Error(`获取用户信息失败: ${data.msg}`);
    77→    }
    78→
    79→    return {
    80→      userId: data.data.user_id,
    81→      openId: data.data.open_id,
    82→      name: data.data.name,
    83→      avatar: data.data.avatar_url,
    84→      email: data.data.email,
    85→    };
    86→  }
    87→
    88→  // 生成授权URL
    89→  getAuthUrl() {
    90→    const params = new URLSearchParams({
    91→      app_id: this.appId,
    92→      redirect_uri: this.redirectUri,
    93→      scope: 'contact:user.base:readonly',
    94→      state: Math.random().toString(36).substring(7),
    95→    });
    96→
    97→    return `https://open.feishu.cn/open-apis/authen/v1/authorize?${params.toString()}`;
    98→  }
    99→
   100→  // 刷新用户token
   101→  async refreshUserToken(refreshToken) {
   102→    const response = await fetch(`${FEISHU_BASE_URL}/authen/v1/oidc/refresh_access_token`, {
   103→      method: 'POST',
   104→      headers: { 'Content-Type': 'application/json' },
   105→      body: JSON.stringify({
   106→        grant_type: 'refresh_token',
   107→        client_id: this.appId,
   108→        client_secret: this.appSecret,
   109→        refresh_token: refreshToken,
   110→      }),
   111→    });
   112→
   113→    const data = await response.json();
   114→    if (data.code !== 0) {
   115→      throw new Error(`刷新token失败: ${data.msg}`);
   116→    }
   117→
   118→    return {
   119→      accessToken: data.access_token,
   120→      refreshToken: data.refresh_token,
   121→      expiresIn: data.expires_in,
   122→    };
   123→  }
   124→}
   125→
   126→module.exports = FeishuAuthClient;
