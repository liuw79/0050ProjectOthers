     1→// 飞书API客户端 - 使用直接HTTP调用
     2→const path = require('path');
     3→require('dotenv').config({ path: path.join(__dirname, '../.env') });
     4→
     5→const FEISHU_BASE_URL = 'https://open.feishu.cn/open-apis';
     6→
     7→class LarkClient {
     8→  constructor() {
     9→    this.appAccessToken = null;
    10→    this.tokenExpireTime = 0;
    11→  }
    12→
    13→  // 获取app access token
    14→  async getAppAccessToken() {
    15→    // 检查token是否过期
    16→    if (this.appAccessToken && Date.now() < this.tokenExpireTime) {
    17→      return this.appAccessToken;
    18→    }
    19→
    20→    try {
    21→      const response = await fetch(`${FEISHU_BASE_URL}/auth/v3/app_access_token/internal`, {
    22→        method: 'POST',
    23→        headers: { 'Content-Type': 'application/json' },
    24→        body: JSON.stringify({
    25→          app_id: process.env.LARK_APP_ID,
    26→          app_secret: process.env.LARK_APP_SECRET,
    27→        }),
    28→      });
    29→
    30→      const data = await response.json();
    31→      if (data.code !== 0) {
    32→        throw new Error(`获取飞书token失败: ${data.msg}`);
    33→      }
    34→
    35→      this.appAccessToken = data.app_access_token;
    36→      this.tokenExpireTime = Date.now() + (data.expire_in - 300) * 1000; // 提前5分钟过期
    37→      return this.appAccessToken;
    38→    } catch (error) {
    39→      console.error('获取飞书token失败:', error);
    40→      throw error;
    41→    }
    42→  }
    43→
    44→  // 获取多维表格数据
    45→  async getBitableData(appToken, tableId, options = {}) {
    46→    try {
    47→      const { page_size = 100, page_token = '' } = options;
    48→      const token = await this.getAppAccessToken();
    49→      const url = `${FEISHU_BASE_URL}/bitable/v1/apps/${appToken}/tables/${tableId}/records?page_size=${page_size}&page_token=${page_token}`;
    50→
    51→      const response = await fetch(url, {
    52→        method: 'GET',
    53→        headers: {
    54→          'Authorization': `Bearer ${token}`,
    55→          'Content-Type': 'application/json',
    56→        },
    57→      });
    58→
    59→      const data = await response.json();
    60→      if (data.code !== 0) {
    61→        throw new Error(`获取多维表格数据失败: ${data.msg}`);
    62→      }
    63→
    64→      return data.data.items;
    65→    } catch (error) {
    66→      console.error('获取多维表格数据失败:', error);
    67→      throw error;
    68→    }
    69→  }
    70→
